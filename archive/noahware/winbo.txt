```MD
# winbo


Detecting window hijacking via ETW and GDI table scanning.


# How does it work?


winbo parses the dxgkrnl.sys ETW events to see which processes call Present on which window handles. The process id of the Present call is compared to the process id of the window owner, and if they do not match then a process is rendering on another process’s window (hijacked). If the processes share a common parent process then they are allowed, as legitimate applications can a cluster of processes sharing one window.

The above check works for when windows are rendered with any major graphics APIs (DirectX, OpenGL, etc). There is one other aspect to cover: GDI. GDI rendering by creating a handle to the window’s DC and copying a draw buffer can also be used to render. To detect this, the shared GDI handle table is iterated to find DC handles. The owner of the DC handle is compared to the Window it points to and if they don’t match, then the Window is being hijacked.
```

```cpp
#include "../win/handle.hpp"

static void event_record_callback(const PEVENT_RECORD event)
{
	constexpr std::uint16_t present_event_id = 0xB8;

	const auto& event_descriptor = event->EventHeader.EventDescriptor;

	if (event_descriptor.Id == present_event_id)
	{
		try
		{
			const etw::trace_event_info event_info(event);
			const win::window window(event_info.property<HWND>(L"hWindow"));

			if (window)
			{
				const win::process present_process(event->EventHeader.ProcessId);
				const win::process window_process = window.process();

				if (window_process.id() && present_process != window_process && present_process.
					parent_recursive_or_this() != window_process.parent_recursive_or_this())
				{
					spdlog::warn("process '{}' ({}) rendering on window of another process '{}' ({})", present_process.id(),
						present_process.image_name(), window_process.id(), window_process.image_name());
				}
			}
		}
		catch (const std::exception& e)
		{
			spdlog::error("caught exception: '{}'", e.what());
		}
	}
}

static void stop_existing_session()
{
	const auto cmd = std::format("logman stop {} -ets >nul", TRACE_SESSION_NAME);

	std::system(cmd.c_str());
}

void etw::start_tracing()
{
	stop_existing_session();

	CONTROLTRACE_ID trace_id = { };
	trace_properties properties = { };

	properties.Wnode.BufferSize = sizeof(properties);
	properties.Wnode.ClientContext = 1;
	properties.Wnode.Flags = WNODE_FLAG_TRACED_GUID;
	properties.LoggerNameOffset = offsetof(trace_properties, session_name);
	properties.LogFileMode = EVENT_TRACE_REAL_TIME_MODE;

	if (FAILED(StartTraceA(&trace_id, TRACE_SESSION_NAME, &properties)))
	{
		throw std::runtime_error("unable to start trace");
	}

	constexpr GUID guid = { // dxgkrnl
		0x802EC45A,
		0x1E99,
		0x4B83,
		{ 0x99, 0x20, 0x87, 0xC9, 0x82, 0x77, 0xBA, 0x9D }
	};

	if (FAILED(EnableTraceEx2(trace_id, &guid, EVENT_CONTROL_CODE_ENABLE_PROVIDER, TRACE_LEVEL_VERBOSE, 0, 0, 0, nullptr)))
	{
		throw std::runtime_error("unable to enable trace");
	}

	EVENT_TRACE_LOGFILEA trace_logfile = { };

	trace_logfile.ProcessTraceMode = PROCESS_TRACE_MODE_EVENT_RECORD | PROCESS_TRACE_MODE_REAL_TIME;
	trace_logfile.LoggerName = const_cast<LPSTR>(TRACE_SESSION_NAME);

	trace_logfile.EventRecordCallback = event_record_callback;

	auto trace_handle = OpenTraceA(&trace_logfile);

	if (trace_handle == INVALID_PROCESSTRACE_HANDLE)
	{
		throw std::runtime_error("unable to open trace");
	}

	if (FAILED(ProcessTrace(&trace_handle, 1, nullptr, nullptr)))
	{
		CloseTrace(trace_handle);

		throw std::runtime_error("unable to process trace");
	}

	CloseTrace(trace_handle);
}

etw::trace_event_info::trace_event_info(const PEVENT_RECORD event_record)
		:	event_record_(event_record)
{
	std::uint32_t buffer_size = 0;

	if (TdhGetEventInformation(event_record_, 0, nullptr, nullptr, reinterpret_cast<PULONG>(&buffer_size)) !=
		ERROR_INSUFFICIENT_BUFFER)
	{
		throw std::runtime_error("unable to query event information size");
	}

	buffer_.resize(buffer_size);

	if (FAILED(
		TdhGetEventInformation(event_record_, 0, nullptr, reinterpret_cast<value_type*>(buffer_.data()),
			reinterpret_cast<PULONG>(&buffer_size))))
	{
		throw std::runtime_error("unable to query event information buffer");
	}
}

etw::trace_event_info::value_type* etw::trace_event_info::value()
{
	return reinterpret_cast<value_type*>(buffer_.data());
}

const etw::trace_event_info::value_type* etw::trace_event_info::value() const
{
	return reinterpret_cast<const value_type*>(buffer_.data());
}

std::span<const EVENT_PROPERTY_INFO> etw::trace_event_info::properties() const
{
	const auto event = value();

	return { event->EventPropertyInfoArray, event->PropertyCount };
}

std::wstring_view etw::trace_event_info::property_name(const EVENT_PROPERTY_INFO& property) const
{
	const auto buffer = TEI_PROPERTY_NAME(const_cast<value_type*>(value()), const_cast<EVENT_PROPERTY_INFO*>(&property));

	if (!buffer)
	{
		return { };
	}

	return { buffer };
}
```

```hpp
#pragma once
#include <Windows.h>
#include <evntrace.h>
#include <evntcons.h>
#include <tdh.h>

#include <string_view>
#include <stdexcept>
#include <optional>
#include <vector>
#include <array>
#include <span>

#define TRACE_SESSION_NAME "winbo"
#define TRACE_SESSION_NAME_SIZE sizeof(TRACE_SESSION_NAME)

namespace etw
{
	void start_tracing();

	struct trace_properties : EVENT_TRACE_PROPERTIES
	{
		std::array<char, TRACE_SESSION_NAME_SIZE> session_name;
	};

	class trace_event_info
	{
	public:
		using value_type = TRACE_EVENT_INFO;

		explicit trace_event_info(PEVENT_RECORD event_record);

		template <class T>
		[[nodiscard]] std::optional<T> property_opt(const std::wstring_view name) const
		{
			std::size_t offset = 0;

			for (const auto& property : properties())
			{
				const auto current_name = property_name(property);
				const std::size_t current_size = property.length;

				if (current_name == name)
				{
					if (sizeof(T) != current_size)
					{
						throw std::runtime_error("event property size != sizeof(T)");
					}

					return *reinterpret_cast<const T*>(static_cast<const std::uint8_t*>(event_record_->UserData) + offset);
				}

				offset += current_size * property.count;
			}

			return std::nullopt;
		}

		template <class T>
		[[nodiscard]] T property(const std::wstring_view name) const
		{
			const auto property = property_opt<T>(name);

			return property.value();
		}

		[[nodiscard]] value_type* value();
		[[nodiscard]] const value_type* value() const;

	protected:
		[[nodiscard]] std::span<const EVENT_PROPERTY_INFO> properties() const;
		[[nodiscard]] std::wstring_view property_name(const EVENT_PROPERTY_INFO& property) const;

		std::vector<std::uint8_t> buffer_;
		PEVENT_RECORD event_record_;
	};

}
```

```cpp
#include "gdi.hpp"
#include <spdlog/spdlog.h>
#include <Windows.h>
#include <intrin.h>
#include <chrono>
#include <thread>
#include <span>

#include "../win/process.hpp"
#include "../win/window.hpp"

std::span<const gdi::handle_entry> get_gdi_handle_entries()
{
	constexpr std::size_t gdi_table_offset = 0xF8;
	constexpr std::size_t gdi_entry_count = 0x4000;

	const std::uint64_t peb_address = __readgsqword(0x60);
	const gdi::handle_entry* const begin = *reinterpret_cast<gdi::handle_entry**>(peb_address + gdi_table_offset);

	return { begin, gdi_entry_count };
}

static void search_gdi_handle_entries()
{
	const auto gdi_handle_entries = get_gdi_handle_entries();

	for (gdi::handle_entry::index_type i = 0; i < gdi_handle_entries.size(); i++)
	{
		const auto& gdi_handle = gdi_handle_entries[i];

		if (gdi_handle.type == gdi::handle_type::dc)
		{
			const gdi::handle_entry::value_type dc_value = gdi_handle.value(i);
			const win::window window(WindowFromDC(reinterpret_cast<HDC>(dc_value)));

			if (window && window.could_be_overlay())
			{
				const win::process owning_process(gdi_handle.process_id);
				const win::process window_process = window.process();

				if (window_process.id() && owning_process.id() != window_process.id() && owning_process.
					parent_recursive_or_this() != window_process.parent_recursive_or_this())
				{
					spdlog::warn("process '{}' ({}) has HDC open to window of another process '{}' ({})",
					             owning_process.id(), owning_process.image_name(), window_process.id(),
					             window_process.image_name());
				}
			}
		}
	}
}

void gdi::start_table_scanning()
{
	while (true)
	{
		search_gdi_handle_entries();

		std::this_thread::sleep_for(std::chrono::milliseconds(150));
	}
}
```

```hpp
#pragma once
#include <cstdint>

namespace gdi
{
	__declspec(noreturn) void start_table_scanning();

	enum class handle_type : std::uint8_t
	{
		dc = 1
	};

	struct handle_entry
	{
		using value_type = std::uint64_t;
		using index_type = std::uint64_t;

		std::uint8_t pad[8];
		std::uint16_t process_id;
		std::uint8_t pad2[2];
		std::uint16_t unique;
		handle_type type;
		std::uint8_t flags;
		std::uint8_t pad3[8];

		[[nodiscard]] value_type value(const index_type table_index) const
		{
			return (unique << 16) | table_index;
		}
	};
}
```

```hpp
#pragma once
#include <Windows.h>

namespace win
{
	class scoped_handle
	{
	public:
		using value_type = HANDLE;

		static const scoped_handle invalid_value;

		explicit scoped_handle(const value_type value) noexcept
				:	value_(value) { }

		scoped_handle(const scoped_handle&) = delete;
		scoped_handle& operator=(const scoped_handle&) = delete;

		scoped_handle(scoped_handle&& right) noexcept
				:	value_(right.value_)
		{
			right.value_ = nullptr;
		}

		scoped_handle& operator=(scoped_handle&& right) noexcept
		{
			if (this != &right)
			{
				close();

				value_ = right.value_;

				right.value_ = nullptr;
			}

			return *this;
		}

		void close() noexcept
		{
			if (*this)
			{
				CloseHandle(value_);

				value_ = nullptr;
			}
		}

		explicit operator bool() const noexcept
		{
			return value_ && *this != invalid_value;
		}

		[[nodiscard]] bool operator==(const scoped_handle& right) const noexcept
		{
			return value_ == right.value_;
		}

		[[nodiscard]] bool operator!=(const scoped_handle& right) const noexcept
		{
			return value_ != right.value_;
		}

		[[nodiscard]] value_type value() const noexcept
		{
			return value_;
		}

		~scoped_handle() noexcept
		{
			close();
		}

	protected:
		value_type value_ = nullptr;
	};

	inline const scoped_handle scoped_handle::invalid_value{ INVALID_HANDLE_VALUE };
}
```

```cpp
#include "process.hpp"
#include "handle.hpp"
#include "nt.hpp"

#include <array>

std::optional<win::process> win::process::parent() const
{
	const scoped_handle process_handle = open_handle(query_limited_information);

	if (!process_handle)
	{
		return std::nullopt;
	}

	PROCESS_BASIC_INFORMATION info = { };

	const nt::status status = nt::query_process_information(process_handle, ProcessBasicInformation, &info, sizeof(info));

	if (!status || !info.Reserved3)
	{
		return std::nullopt;
	}

	const id_type parent_process_id = static_cast<id_type>(reinterpret_cast<std::uint64_t>(info.Reserved3));
	const process parent_process{ parent_process_id };

	if (parent_process.creation_time() < creation_time())
	{
		return std::nullopt;
	}

	return parent_process;
}

std::optional<win::process> win::process::parent_recursive() const
{
	std::optional<process> current_parent = parent();
	std::optional<process> last_parent = std::nullopt;

	while (current_parent)
	{
		last_parent = current_parent;

		current_parent = current_parent->parent();
	}

	return last_parent;
}

win::process win::process::parent_recursive_or_this() const
{
	const auto resulting_parent = parent_recursive();

	return resulting_parent.value_or(*this);
}

win::process::time_type win::process::creation_time() const noexcept
{
	const scoped_handle process_handle = open_handle(query_limited_information);

	FILETIME creation_time = { };
	FILETIME unused_time = { };

	if (!GetProcessTimes(process_handle.value(), &creation_time, &unused_time, &unused_time, &unused_time))
	{
		return 0;
	}

	return unused_time.dwLowDateTime | static_cast<std::uint64_t>(unused_time.dwHighDateTime) << 32;
}

std::string win::process::image_name() const
{
	constexpr std::uint32_t path_length = MAX_PATH;

	const scoped_handle process_handle = open_handle(query_limited_information);

	if (!process_handle)
	{
		return { };
	}

	std::uint32_t actual_length = path_length;

	std::array<char, path_length> buffer;

	if (QueryFullProcessImageNameA(process_handle.value(), 0, buffer.data(), reinterpret_cast<PDWORD>(&actual_length)))
	{
		return { buffer.data(), actual_length };
	}

	return { };
}

win::process::id_type win::process::id() const noexcept
{
	return id_;
}

bool win::process::operator==(const process& right) const noexcept
{
	return id_ == right.id_;
}

bool win::process::operator!=(const process& right) const noexcept
{
	return id_ != right.id_;
}

win::scoped_handle win::process::open_handle(const access_type access) const
{
	return scoped_handle{ OpenProcess(access, false, id_) };
}
```

```hpp
#pragma once
#include "process.hpp"
#include "handle.hpp"
#include <Windows.h>
#include <cstdint>

namespace win
{
	class process;

	class window
	{
	public:
		using handle_type = HWND;

		explicit window(const handle_type handle) noexcept
				:	handle_(handle) { }

		[[nodiscard]] process process() const noexcept
		{
			std::uint32_t process_id = 0;

			GetWindowThreadProcessId(handle_, reinterpret_cast<LPDWORD>(&process_id));

			return win::process{ process_id };
		}

		[[nodiscard]] bool could_be_overlay() const noexcept
		{
			const auto ex_style = GetWindowLongPtr(handle_, GWL_EXSTYLE);

			return ex_style & WS_EX_TOPMOST || ex_style & WS_EX_TRANSPARENT;
		}

		explicit operator bool() const noexcept
		{
			return handle_ && handle_ != scoped_handle::invalid_value.value() && IsWindow(handle_);
		}

	protected:
		handle_type handle_;
	};
}
```

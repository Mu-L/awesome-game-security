all: $(KEY_FILE)
	$(MAKE) -C $(KDIR) M=$(PWD) modules
	@echo ""
	@echo "╔═══════════════════════════════════════════════════════════════╗"
	@echo "║           ksentinel compiled successfully!                    ║"
	@echo "╚═══════════════════════════════════════════════════════════════╝"
	@echo ""

$(KEY_FILE):
	@echo ""
	@echo "╔═══════════════════════════════════════════════════════════════╗"
	@echo "║              GENERATING RANDOM UNLOCK KEY                     ║"
	@echo "╚═══════════════════════════════════════════════════════════════╝"
	@echo ""
	@KEY=$$(head -c 16 /dev/urandom | od -A n -t x1 | tr -d ' \n' | head -c 32); \
	echo "#ifndef KSENTINEL_KEY_H"          >  $(KEY_FILE); \
	echo "#define KSENTINEL_KEY_H"          >> $(KEY_FILE); \
	echo "#define UNLOCK_KEY \"$$KEY\""     >> $(KEY_FILE); \
	echo "#endif"                           >> $(KEY_FILE); \
	echo ""; \
	echo ""; \
	echo "  UNLOCK KEY: $$KEY"; \
	echo ""; \
	echo "════════════════════════════════════════════════════════════════"; \
	echo "  This key is embedded in the module and NOT saved anywhere."; \
	echo "  To unhide the module later: echo \"KEY\" > /proc/.ksctl"; \
	echo "  Or use: make unlock (and paste the key when prompted)"; \
	echo "════════════════════════════════════════════════════════════════"; \
	echo ""; \
	read -p "Press ENTER after you have written down the key..." dummy

unlock:
	@echo ""
	@echo "╔═══════════════════════════════════════════════════════════════╗"
	@echo "║              UNLOCK HIDDEN MODULE                             ║"
	@echo "╚═══════════════════════════════════════════════════════════════╝"
	@echo ""
	@if [ ! -f /proc/.ksctl ]; then \
		echo "[!] Error: /proc/.ksctl not found. Is the module loaded?"; \
		exit 1; \
	fi
	@bash -c 'read -sp "Enter unlock key: " KEY; echo; \
	if [ -z "$$KEY" ]; then \
		echo "[!] Error: key cannot be empty"; \
		exit 1; \
	fi; \
	echo "[*] Writing key to /proc/.ksctl..."; \
	if echo "$$KEY" | sudo tee /proc/.ksctl > /dev/null 2>&1; then \
		sleep 1; \
		if lsmod | grep -q ksentinel; then \
			echo "[+] SUCCESS! Module is now visible"; \
			lsmod | grep ksentinel; \
		else \
			echo "[!] Key written, but module still hidden. Check dmesg."; \
		fi \
	else \
		echo "[!] Failed to write to /proc/.ksctl"; \
	fi'

install: all
	@echo "[*] Loading ksentinel module..."
	sudo insmod ksentinel.ko
	@echo "[+] Module loaded! It is now hidden from lsmod."
	@echo ""
	@echo "To unhide it later:"
	@echo "  1. echo \"YOUR_KEY\" > /proc/.ksctl"

		exit 1; \
	fi

	rm -f $(KEY_FILE)

status:

			echo "[X] Module: NOT LOADED"; \

	fi
	@echo ""
	@if [ -f /proc/.ksctl ]; then \

	fi

help:

`README.md`:

```md
# ksentinel - Kernel Syscalls Integrity Monitor

<img src="https://i.imgur.com/iUDmCVJ.png" alt="Singularity Rootkit" width="600"/>

## Overview

ksentinel monitors critical kernel functions and the syscall table for unauthorized modifications. It detects common syscalls that are targeted by most rootkits including ftrace hooks, kprobes, and syscall table hijacking.

## Features

- Function prologue integrity checking (FNV-1a hash-based)
- System call table monitoring (512 entries)
- LSTAR MSR verification (x86_64 syscall entry point)
- Anti-unload protection with unlock key mechanism
- Continuous monitoring with configurable intervals

NOTE: ksentinel is in its initial version; more improvements will come in future commits.

## Monitored Functions

**VFS Layer:**
- filldir, filldir64, do_getdents64, vfs_read, vfs_write, vfs_open

**Network Stack:**
- tcp4_seq_show, tcp6_seq_show, udp4_seq_show, udp6_seq_show

**Credentials:**
- commit_creds, prepare_kernel_cred

**Module System:**
- load_module, free_module

**Tracing Infrastructure:**
- register_kprobe, register_ftrace_function, ftrace_set_filter

Plus 500+ additional syscall wrappers via prefix matching.

## Building

```bash
make
```

During compilation, a random 32-character unlock key is generated and displayed. Write it down immediately.
This key is embedded in the compiled module and is not stored externally.

Keep it secure if anti-unload protection is enabled.


## Installation

```bash
sudo make install
```

The module loads hidden from lsmod with anti-unload protection enabled by default.

## Usage

### Manual Loading

```bash
sudo insmod ksentinel.ko
```

### Using Manager Script

```bash
sudo ./ksentinel-manager.sh load      # load module
sudo ./ksentinel-manager.sh status    # check status
sudo ./ksentinel-manager.sh watch     # monitor live
sudo ./ksentinel-manager.sh violations # view alerts
```

### Monitoring Logs

```bash
dmesg -wT | grep ksentinel
```

### Unlocking Module

When anti-unload is enabled, the module must be unlocked before removal:

```bash
echo "your_32char_key" > /proc/.ksctl
```

Or use the manager:

```bash
sudo ./ksentinel-manager.sh unlock
```

### Removing Module

```bash
sudo rmmod ksentinel
```

## Module Parameters

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| period_ms | uint | 2000 | Integrity check interval in milliseconds |
| sym_prefixes | string | "sys_,__x64_sys_,do_sys_,ksys_" | Comma-separated symbol prefixes to monitor |
| extra_syms | string | "" | Additional specific symbols to watch |
| anti_unload | bool | true | Enable anti-unload protection |

Example with custom parameters:

```bash
sudo insmod ksentinel.ko period_ms=5000 anti_unload=false
```

## Performance Considerations

ksentinel performs periodic integrity checks based on the `period_ms` parameter.

Lower intervals provide faster detection but increase CPU usage.
Default interval (2000ms) has minimal impact on modern systems.


## Detection Examples

### Inline Hook Detected

```
ksentinel: [CHANGE - HOOK PATTERN] sys_getdents64 @ 0xffffffff81234567
  baseline=1234567890abcdef  current=fedcba0987654321
  bytes: e9 ab cd ef 12 90 90 90 ...
  flags: JMP/CALL_DETECTED ENDBR64_PREFIX
```

### Syscall Table Hook

```
ksentinel: [SCT HOOK] syscall[78] expected=0xffffffff81111111 got=0xffffffffa0222222 (fake_getdents+0x0)
```

### LSTAR Tampering

```
ksentinel: [LSTAR TAMPERED] baseline=0xffffffff81800000 now=0xffffffffa0300000 (evil_syscall_entry+0x0)
```

## Known Rootkit Detection

ksentinel can detect modifications made by:

- **PUMAKIT** - ftrace-based hooking (18 syscalls)
- **Diamorphine** - syscall table patching
- **KoviD** - kprobes-based hiding

Detection depends on loading ksentinel before the rootkit.

## Requirements

**Kernel Configuration:**
- CONFIG_KALLSYMS=y
- CONFIG_KPROBES=y
- CONFIG_MODULES=y

**Supported Kernels:**
- Linux 5.4 through 6.12+
- x86_64 architecture (LSTAR checking)
- ARM64 support (without LSTAR checks)

## Limitations

1. **Detection Only** - Does not prevent or remove rootkits
2. **Baseline Assumption** - Assumes kernel is clean at module load time
3. **Race Condition** - Rootkits loaded before ksentinel cannot be detected
4. **Evasion** - Sophisticated rootkits can target monitoring code itself

## Security Considerations

- Load ksentinel as early as possible during boot
- Store unlock key securely offline

## Troubleshooting

### Module Won't Load

Check kernel configuration:
```bash
grep -E "KALLSYMS|KPROBES" /boot/config-$(uname -r)
```

### Module Won't Unload

The module is hidden with anti-unload protection. Unlock first:
```bash
sudo ./ksentinel-manager.sh unlock
```

#!/bin/bash
#
# ksentinel-manager.sh - Kernel integrity monitor controller
# Manages the ksentinel LKM
#

set -euo pipefail

PROC_CONTROL="/proc/.ksctl"
MODULE_NAME="ksentinel"

if command -v tput >/dev/null 2>&1 && [ -t 1 ]; then
    RED=$(tput setaf 1)
    GREEN=$(tput setaf 2)
    YELLOW=$(tput setaf 3)
    BLUE=$(tput setaf 4)
    CYAN=$(tput setaf 6)
    BOLD=$(tput bold)
    RESET=$(tput sgr0)
else
    RED='\033[0;31m'
    GREEN='\033[0;32m'
    YELLOW='\033[1;33m'
    BLUE='\033[0;34m'
    CYAN='\033[0;36m'

get_refcount() {
    if is_visible; then

get_stats() {
    local sym_count=$(dmesg | grep "$MODULE_NAME" | grep -oP '\d+(?= symbols watched)' | tail -1 | tr -d '[:space:]')
    local sct_count=$(dmesg | grep "$MODULE_NAME" | grep -oP '\d+(?= SCT entries)' | tail -1 | tr -d '[:space:]')
    local sensitive=$(dmesg | grep "$MODULE_NAME" | grep -oP '\d+(?= sensitive symbols)' | tail -1 | tr -d '[:space:]')
    local prefix=$(dmesg | grep "$MODULE_NAME" | grep -oP '\d+(?= prefix-matched)' | tail -1 | tr -d '[:space:]')
    
    echo "${sym_count:-?}:${sct_count:-?}:${sensitive:-?}:${prefix:-?}"
}

count_violations() {

cmd_load() {

        echo
        echo -e "${GREEN}✓${RESET} Module loaded successfully"
        echo
        
        if is_visible; then
            echo -e "  State: ${GREEN}visible${RESET} (protection disabled)"
        else
            echo -e "  State: ${CYAN}hidden${RESET} (anti-unload active)"
            echo
            echo -e "${YELLOW}Note:${RESET} The module has automatically hidden itself."
            echo -e "      To unload later, you must first unlock it:"

            echo -e "        ${BOLD}1.${RESET} ${CYAN}$0 unlock${RESET}  ${YELLOW}# enter the unlock key${RESET}"
            echo -e "        ${BOLD}2.${RESET} ${CYAN}$0 unload${RESET}"
        fi
        
        echo
        echo -e "${BLUE}→${RESET} Run '${CYAN}$0 status${RESET}' to view monitoring details"
    else
        echo
        echo -e "${RED}✗${RESET} Failed to load module"

static notrace int __init resolve_unexported_syms(void)
{
	struct kprobe kp = { .symbol_name = "kallsyms_lookup_name" };
	int ret;

	ret = register_kprobe(&kp);
	if (ret < 0) {

		return ret;

	unregister_kprobe(&kp);

		pr_err("ksentinel: failed to resolve kallsyms_lookup_name\n");
		return -ENOENT;

static u64 notrace fnv1a64(const void *data, size_t len)
{
	const u8 *p = data;
	u64 h = 14695981039346656037ULL;
	size_t i;

	for (i = 0; i < len; i++) {
		h ^= (u64)p[i];
		h *= 1099511628211ULL;
	}
	return h;
}

static bool notrace buf_starts_with_endbr64(const u8 *buf)
{
	return (buf[0] == ENDBR64_B0 && buf[1] == ENDBR64_B1 &&
		buf[2] == ENDBR64_B2 && buf[3] == ENDBR64_B3);
}

static bool notrace prologue_looks_hooked(const u8 *buf)
{
	int off = buf_starts_with_endbr64(buf) ? ENDBR64_LEN : 0;
	const u8 *p = buf + off;

	if (off + 2 > PROLOGUE_BYTES)
		return false;
	if (p[0] == OPCODE_JMP_REL32)
		return true;
	if (p[0] == OPCODE_CALL_REL32)
		return true;
	if (p[0] == OPCODE_JMP_IND_HI && p[1] == OPCODE_JMP_IND_LO)
		return true;
	return false;
}

struct sym_entry {
	char          name[KSYM_NAME_LEN];
	unsigned long addr;
	u64           hash;
	u8            prologue[PROLOGUE_BYTES];
	bool          valid;
	bool          alerted;
};

static struct sym_entry *watchlist;
static int               watchlist_count;
static DEFINE_MUTEX(watchlist_mutex);

static bool notrace watchlist_add(const char *name, unsigned long addr)
{
	struct sym_entry *e;
	u8 buf[PROLOGUE_BYTES];
	int i;

	if (!addr || watchlist_count >= MAX_WATCH_SYMS)
		return false;

	for (i = 0; i < watchlist_count; i++) {
		if (watchlist[i].addr == addr)
			return false;
	}

	if (copy_from_kernel_nofault(buf, (const void *)addr, PROLOGUE_BYTES) != 0)
		return false;

	e = &watchlist[watchlist_count];
	strscpy(e->name, name, sizeof(e->name));
	e->addr  = addr;
	memcpy(e->prologue, buf, PROLOGUE_BYTES);
	e->hash  = fnv1a64(buf, PROLOGUE_BYTES);
	e->valid = true;
	e->alerted = false;
	watchlist_count++;
	return true;
}

struct iter_ctx {
	char *prefixes[32];
	int   n_prefixes;
};

static bool notrace matches_prefix(const char *name, const struct iter_ctx *ctx)
{
	int i;

	for (i = 0; i < ctx->n_prefixes; i++) {
		if (strncmp(name, ctx->prefixes[i], strlen(ctx->prefixes[i])) == 0)
			return true;
	}
	return false;
}

static int notrace sym_iter_cb_impl(void *data, const char *name, unsigned long addr)
{
	if (watchlist_count >= MAX_WATCH_SYMS)
		return 1;
	if (matches_prefix(name, data))
		watchlist_add(name, addr);
	return 0;
}

#if LINUX_VERSION_CODE >= KERNEL_VERSION(6, 1, 0)
static int notrace sym_iter_cb(void *data, const char *name, unsigned long addr)
{
	return sym_iter_cb_impl(data, name, addr);
}
#else
static int notrace sym_iter_cb(void *data, const char *name,
		       struct module *mod __maybe_unused, unsigned long addr)
{
	return sym_iter_cb_impl(data, name, addr);
}
#endif

static void notrace parse_prefixes(struct iter_ctx *ctx, const char *param,
			   char *buf, size_t bufsz)
{
	char *tok, *pos;

	strscpy(buf, param, bufsz);
	pos = buf;
	ctx->n_prefixes = 0;
	while ((tok = strsep(&pos, ",")) != NULL &&
	       ctx->n_prefixes < (int)ARRAY_SIZE(ctx->prefixes)) {
		if (*tok)
			ctx->prefixes[ctx->n_prefixes++] = tok;
	}
}

static notrace int __init build_watchlist(void)
{
	struct iter_ctx ctx = {};
	char prefix_buf[256];
	char extra_buf[512];
	const char * const *p;
	int added_sensitive = 0, added_prefix, added_extra = 0;

	watchlist = kcalloc(MAX_WATCH_SYMS, sizeof(*watchlist), GFP_KERNEL);
	if (!watchlist)
		return -ENOMEM;

	for (p = sensitive_syms; *p; p++) {
		unsigned long addr = ksg_lookup_name(*p);

		if (addr && watchlist_add(*p, addr))
			added_sensitive++;
	}
	pr_info("ksentinel: %d sensitive symbols resolved\n", added_sensitive);

	parse_prefixes(&ctx, sym_prefixes, prefix_buf, sizeof(prefix_buf));
	if (ctx.n_prefixes > 0) {
		int before = watchlist_count;

		ksg_on_each_symbol(sym_iter_cb, &ctx);
		added_prefix = watchlist_count - before;
		pr_info("ksentinel: %d prefix-matched symbols added\n", added_prefix);
	}

	if (extra_syms && *extra_syms) {
		char *tok, *pos;

		strscpy(extra_buf, extra_syms, sizeof(extra_buf));
		pos = extra_buf;
		while ((tok = strsep(&pos, ",")) != NULL) {
			unsigned long addr;

			if (!*tok)
				continue;
			addr = ksg_lookup_name(tok);
			if (!addr) {
				pr_warn("ksentinel: extra sym '%s' not found\n", tok);
				continue;
			}
			if (watchlist_add(tok, addr))
				added_extra++;
		}
		if (added_extra)
			pr_info("ksentinel: %d extra symbols added\n", added_extra);
	}

	pr_info("ksentinel: total watchlist = %d symbols\n", watchlist_count);
	return 0;
}

static void **sys_call_table_ptr;
static unsigned long sct_baseline[NR_SYSCALLS_WATCH];
static int sct_count;

static notrace void __init sct_build_baseline(void)
{
	int i;
	unsigned long entry;

	sys_call_table_ptr = (void **)ksg_lookup_name("sys_call_table");
	if (!sys_call_table_ptr) {
		pr_warn("ksentinel: sys_call_table not found, skipping SCT checks\n");
		return;
	}

	for (i = 0; i < NR_SYSCALLS_WATCH; i++) {
		if (copy_from_kernel_nofault(&entry, &sys_call_table_ptr[i],
					     sizeof(entry)) != 0)
			break;
		sct_baseline[i] = entry;
		sct_count++;
	}
	pr_info("ksentinel: sys_call_table @ %pK — %d entries baselined\n",
		sys_call_table_ptr, sct_count);
}

static void notrace sct_check(void)
{
	int i;
	unsigned long now;

	if (!sys_call_table_ptr || !sct_count)
		return;

	for (i = 0; i < sct_count; i++) {
		if (copy_from_kernel_nofault(&now, &sys_call_table_ptr[i],
					     sizeof(now)) != 0)
			continue;
		if (now != sct_baseline[i]) {
			char sym[KSYM_SYMBOL_LEN] = {};

			sprint_symbol(sym, now);
			pr_err("ksentinel: [SCT HOOK] syscall[%3d] expected=%pK got=%pK (%s)\n",
			       i, (void *)sct_baseline[i], (void *)now, sym);
		}
	}
}

#ifdef CONFIG_X86_64
#define MSR_LSTAR_ADDR 0xC0000082UL
static unsigned long lstar_baseline;

static notrace void __init lstar_snapshot(void)
{
	rdmsrl(MSR_LSTAR_ADDR, lstar_baseline);
	pr_info("ksentinel: LSTAR baseline = %pK\n", (void *)lstar_baseline);
}

static void notrace lstar_check(void)
{
	unsigned long now;
	char sym[KSYM_SYMBOL_LEN] = {};

	rdmsrl(MSR_LSTAR_ADDR, now);
	if (now != lstar_baseline) {
		sprint_symbol(sym, now);
		pr_err("ksentinel: [LSTAR TAMPERED] baseline=%pK now=%pK (%s)\n",
		       (void *)lstar_baseline, (void *)now, sym);
	}
}
#else
static inline void notrace lstar_snapshot(void) {}
static inline void notrace lstar_check(void)    {}
#endif

struct module_sect_attr {
	struct bin_attribute battr;
	unsigned long address;
};

struct module_sect_attrs {
	struct attribute_group grp;
	unsigned int nsections;
	struct module_sect_attr attrs[];
};

struct restore_info {
	struct kobject *parent;
	struct module_sect_attrs *attrs;
	const char *name;
};

static struct list_head      *saved_prev_module;
static struct restore_info   ksg_restore_info;
static struct proc_dir_entry *ksctl_proc;

static ssize_t notrace show_refcnt(struct module_attribute *mattr,
			   struct module_kobject *mk, char *buffer)
{
	return sprintf(buffer, "%i\n", module_refcount(mk->mod));
}

static struct module_attribute modinfo_refcnt =
	__ATTR(refcnt, 0444, show_refcnt, NULL);

static struct module_attribute *modinfo_attrs[] = {
	&modinfo_refcnt,
	NULL,
};

static void notrace module_remove_modinfo_attrs(struct module *mod)
{
	struct module_attribute *attr;

	attr = &mod->modinfo_attrs[0];
	if (attr && attr->attr.name) {
		sysfs_remove_file(&mod->mkobj.kobj, &attr->attr);
		if (attr->free)
			attr->free(mod);
	}
	kfree(mod->modinfo_attrs);
}

static int notrace module_add_modinfo_attrs(struct module *mod)
{
	struct module_attribute *attr;
	struct module_attribute *temp_attr;
	int error = 0;

	mod->modinfo_attrs = kzalloc(
		(sizeof(struct module_attribute) * (ARRAY_SIZE(modinfo_attrs) + 1)),
		GFP_KERNEL);
	if (!mod->modinfo_attrs)
		return -ENOMEM;

	temp_attr = mod->modinfo_attrs;
	attr = modinfo_attrs[0];
	if (!attr->test || attr->test(mod)) {
		memcpy(temp_attr, attr, sizeof(*temp_attr));
		sysfs_attr_init(&temp_attr->attr);
		error = sysfs_create_file(&mod->mkobj.kobj, &temp_attr->attr);
		if (error)
			goto error_out;
	}

	return 0;

error_out:
	module_remove_modinfo_attrs(mod);
	return error;
}

static void notrace sys_hide(void)
{
	ksg_restore_info.attrs = THIS_MODULE->sect_attrs;
	ksg_restore_info.parent = THIS_MODULE->mkobj.kobj.parent;
	ksg_restore_info.name = THIS_MODULE->name;

	kobject_del(THIS_MODULE->holders_dir->parent);

	THIS_MODULE->holders_dir->parent->state_in_sysfs = 1;
	THIS_MODULE->state = MODULE_STATE_UNFORMED;
}

static void notrace sys_show(void)
{
	int err;
	struct kobject *kobj;

	THIS_MODULE->state = MODULE_STATE_LIVE;

	err = kobject_add(&THIS_MODULE->mkobj.kobj, ksg_restore_info.parent,
			  ksg_restore_info.name);
	if (err) {
		pr_err("ksentinel: failed to restore kobject: %d\n", err);
		goto cleanup;
	}

	kobj = kobject_create_and_add("holders", &THIS_MODULE->mkobj.kobj);
	if (!kobj) {
		pr_err("ksentinel: failed to create holders dir\n");
		goto cleanup;
	}
	THIS_MODULE->holders_dir = kobj;

	if (ksg_restore_info.attrs) {
		err = sysfs_create_group(&THIS_MODULE->mkobj.kobj,
					 &ksg_restore_info.attrs->grp);
		if (err) {
			pr_err("ksentinel: failed to restore sysfs group: %d\n", err);
			goto cleanup;
		}
	}

	err = module_add_modinfo_attrs(THIS_MODULE);
	if (err)
		pr_warn("ksentinel: failed to restore modinfo attrs: %d\n", err);

cleanup:
	kobject_put(&THIS_MODULE->mkobj.kobj);
	return;
}

static void notrace unprotect_from_unload(void)
{
	if (!saved_prev_module)
		return;

	mutex_lock(ksg_module_mutex);
	list_add(&THIS_MODULE->list, saved_prev_module);
	mutex_unlock(ksg_module_mutex);

	sys_show();

	module_put(THIS_MODULE);
	
	saved_prev_module = NULL;

	pr_info("ksentinel: [anti-unload] removed — module now visible and unloadable\n");
}

static ssize_t notrace ksctl_write(struct file *file, const char __user *ubuf,
			   size_t count, loff_t *ppos)
{
	char buf[65] = {};
	size_t len;

	len = min(count, sizeof(buf) - 1);
	if (copy_from_user(buf, ubuf, len))
		return -EFAULT;

	if (len > 0 && buf[len - 1] == '\n')
		buf[len - 1] = '\0';

	if (strcmp(buf, UNLOCK_KEY) == 0) {
		pr_info("ksentinel: correct key — unhiding module\n");
		unprotect_from_unload();
	} else {
		pr_warn("ksentinel: wrong key attempt on /proc/%s\n", KSG_PROC_NAME);
	}

	return count;
}

#if LINUX_VERSION_CODE >= KERNEL_VERSION(5, 6, 0)
static const struct proc_ops ksctl_fops = {
	.proc_write = ksctl_write,
};
#else
static const struct file_operations ksctl_fops = {
	.owner = THIS_MODULE,
	.write = ksctl_write,
};
#endif

static void notrace protect_from_unload(void)
{
	if (!anti_unload)
		return;

	try_module_get(THIS_MODULE);

	mutex_lock(ksg_module_mutex);
	saved_prev_module = THIS_MODULE->list.prev;
	list_del(&THIS_MODULE->list);
	INIT_LIST_HEAD(&THIS_MODULE->list);
	mutex_unlock(ksg_module_mutex);

	sys_hide();

	pr_info("ksentinel: [anti-unload] active — hidden from module list and sysfs\n");
}

static struct delayed_work guard_work;

static void notrace guard_tick(struct work_struct *work)
{
	u8 buf[PROLOGUE_BYTES];
	int i, violations = 0;

	mutex_lock(&watchlist_mutex);

	for (i = 0; i < watchlist_count; i++) {
		struct sym_entry *e = &watchlist[i];
		u64 now_hash;

		if (!e->valid || !e->addr)
			continue;

		if (copy_from_kernel_nofault(buf, (const void *)e->addr,
					     PROLOGUE_BYTES) != 0) {
			pr_warn("ksentinel: unreadable: %s @ %pK\n",
				e->name, (void *)e->addr);
			continue;
		}

		now_hash = fnv1a64(buf, PROLOGUE_BYTES);
		if (now_hash != e->hash) {
			if (!e->alerted) {
				bool hooked    = prologue_looks_hooked(buf);
				bool has_endbr = buf_starts_with_endbr64(buf);

				pr_err("ksentinel: [CHANGE%s] %s @ %pK\n"
				       "  baseline=%016llx  current=%016llx\n"
				       "  bytes: %16ph\n"
				       "  flags:%s%s\n",
				       hooked ? " — HOOK PATTERN" : "",
				       e->name, (void *)e->addr,
				       (unsigned long long)e->hash,
				       (unsigned long long)now_hash,
				       buf,
				       hooked    ? " JMP/CALL_DETECTED" : "",
				       has_endbr ? " ENDBR64_PREFIX"    : "");
				e->alerted = true;
			}
			violations++;
		} else {
			if (e->alerted) {
				pr_info("ksentinel: [RESTORED] %s @ %pK returned to baseline\n",
					e->name, (void *)e->addr);
				e->alerted = false;
			}
		}
	}

	mutex_unlock(&watchlist_mutex);

	sct_check();
	lstar_check();

	if (!violations)
		pr_debug("ksentinel: tick OK (%d symbols checked)\n", watchlist_count);

	schedule_delayed_work(&guard_work, msecs_to_jiffies(READ_ONCE(period_ms)));
}

static notrace int __init ksentinel_init(void)
{
	int ret;

	pr_info("ksentinel Beta Testing: starting (period=%u ms, kernel %d.%d, anti_unload=%s)\n",
		period_ms,
		(int)LINUX_VERSION_CODE >> 16,
		((int)LINUX_VERSION_CODE >> 8) & 0xFF,
		anti_unload ? "ON" : "OFF");

	ret = resolve_unexported_syms();
	if (ret)
		return ret;

	ret = build_watchlist();
	if (ret)
		return ret;

	sct_build_baseline();
	lstar_snapshot();

	if (anti_unload) {
		ksctl_proc = proc_create(KSG_PROC_NAME, 0200, NULL, &ksctl_fops);
		if (!ksctl_proc)
			pr_warn("ksentinel: failed to create /proc/%s\n", KSG_PROC_NAME);
	}

	INIT_DELAYED_WORK(&guard_work, guard_tick);
	schedule_delayed_work(&guard_work, msecs_to_jiffies(period_ms));

	pr_info("ksentinel: ready — %d symbols watched, %d SCT entries\n",
		watchlist_count, sct_count);

	protect_from_unload();

	return 0;
}

static notrace void __exit ksentinel_exit(void)
{
	cancel_delayed_work_sync(&guard_work);

	if (ksctl_proc) {
		proc_remove(ksctl_proc);
		ksctl_proc = NULL;
	}

	mutex_lock(&watchlist_mutex);
	kfree(watchlist);
	watchlist       = NULL;
	watchlist_count = 0;
	mutex_unlock(&watchlist_mutex);

	pr_info("ksentinel: unloaded\n");
}

module_init(ksentinel_init);
module_exit(ksentinel_exit);
